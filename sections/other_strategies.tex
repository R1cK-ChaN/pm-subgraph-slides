\begin{frame}{Microstructure-Based Trading Strategies}
    \begin{block}{Strategy Premise}
        \small
        Exploit short-term market mechanics: order flow patterns, price spikes, mean reversion
    \end{block}

    \vspace{0.1cm}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{exampleblock}{Key Signals}
                \small
                \begin{itemize}
                    \item Large trade price impact
                    \item Order flow imbalance
                    \item Volatility spikes
                    \item Volume bursts
                \end{itemize}
            \end{exampleblock}
        \end{column}

        \begin{column}{0.48\textwidth}
            \begin{block}{Data Requirements}
                \small
                \textbf{Available:}
                \begin{itemize}
                    \item[\checkmark] Trade-by-trade data
                    \item[\checkmark] Taker/maker ID
                \end{itemize}
                \textbf{Missing:}
                \begin{itemize}
                    \item[$\times$] Real-time book depth
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.15cm}

    \begin{alertblock}{Feasibility}
        \small
        \textbf{Backtesting:} Moderate $\bullet$ \textbf{Automation:} Requires CLOB API
    \end{alertblock}
\end{frame}

\begin{frame}{Semantic Correlation Arbitrage}
    \begin{block}{Strategy Premise}
        \small
        Exploit relationships between markets: cross-market arbitrage, event correlation
    \end{block}

    \begin{exampleblock}{Example}
        \footnotesize
        Election markets: Candidate X win (80\%) vs Party control (60\%) $\rightarrow$ Mispricing opportunity
    \end{exampleblock}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{Data Coverage}
                \footnotesize
                \textbf{Subgraph:} Price series, correlation

                \textbf{External:} Gamma API, news feeds
            \end{block}
        \end{column}

        \begin{column}{0.48\textwidth}
            \begin{alertblock}{Implementation}
                \footnotesize
                Requires \textbf{SLO} to: map markets, process news, detect implications
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}
