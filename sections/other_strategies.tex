\begin{frame}{Microstructure-Based Trading Strategies}
    \begin{block}{Strategy Premise}
        Exploit short-term market mechanics: order flow patterns, price spikes, mean reversion (seconds to minutes timeframe)
    \end{block}

    \vspace{0.2cm}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{exampleblock}{Key Signals}
                \begin{itemize}
                    \item Large trade price impact
                    \item Order flow imbalance
                    \item Short-term volatility spikes
                    \item Volume bursts
                    \item "Revert-on-spike" patterns
                \end{itemize}
            \end{exampleblock}
        \end{column}

        \begin{column}{0.48\textwidth}
            \begin{block}{Data Requirements}
                \textbf{Available:}
                \begin{itemize}
                    \item[\checkmark] Trade-by-trade data
                    \item[\checkmark] Taker/maker identification
                    \item[\checkmark] Timestamps \& sizes
                \end{itemize}
                \textbf{Missing:}
                \begin{itemize}
                    \item[$\times$] Real-time order book depth
                    \item[$\times$] Sub-second granularity
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.3cm}

    \begin{alertblock}{Feasibility}
        \textbf{Backtesting:} Moderate (can detect patterns, limited by order book data) \\
        \textbf{Automation:} Requires CLOB API integration for live execution
    \end{alertblock}
\end{frame}

\begin{frame}{Semantic Correlation Arbitrage}
    \begin{block}{Strategy Premise}
        Exploit relationships between markets or external events: cross-market arbitrage, news-driven trades, event correlation
    \end{block}

    \vspace{0.2cm}

    \begin{exampleblock}{Example: Election Market Correlation}
        \begin{itemize}
            \item Market A: ``Will Candidate X win presidency?''
            \item Market B: ``Which party controls Senate?''
            \item \textbf{Opportunity:} If Market A implies 80\% chance for Party Y, but Market B only prices 60\% for Party Y control $\rightarrow$ \textbf{Arbitrage}
        \end{itemize}
    \end{exampleblock}

    \vspace{0.2cm}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{Data Coverage}
                \textbf{Subgraph provides:}
                \begin{itemize}
                    \item[\checkmark] Price series per market
                    \item[\checkmark] Cross-market correlation analysis
                \end{itemize}
                \textbf{External needed:}
                \begin{itemize}
                    \item[$\rightarrow$] Gamma API (metadata, groupings)
                    \item[$\rightarrow$] News feeds / NLP
                \end{itemize}
            \end{block}
        \end{column}

        \begin{column}{0.48\textwidth}
            \begin{alertblock}{Implementation}
                Requires \textbf{Semantic Logic Oracle (SLO)} to:
                \begin{itemize}
                    \item Map related markets
                    \item Process news signals
                    \item Detect logical implications
                \end{itemize}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Long-Horizon \& ML-Based Strategies}
    \begin{block}{Strategy Premise}
        Operate on days/weeks timeframe: trend-following, predictive models, portfolio optimization
    \end{block}

    \vspace{0.2cm}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{exampleblock}{Approach Examples}
                \begin{itemize}
                    \item \textbf{Trend-following:} Momentum signals, multi-day patterns
                    \item \textbf{ML Prediction:} Train models on historical outcomes
                    \item \textbf{Portfolio:} Diversified positions across markets
                    \item \textbf{Fundamental:} External data integration (polls, stats)
                \end{itemize}
            \end{exampleblock}
        \end{column}

        \begin{column}{0.48\textwidth}
            \begin{block}{ML Training Data}
                \textbf{Subgraph provides:}
                \begin{itemize}
                    \item[\checkmark] 3+ years historical prices
                    \item[\checkmark] \textbf{Outcome labels} (\texttt{Market.winningOutcome})
                    \item[\checkmark] Volume, user activity
                    \item[\checkmark] Daily platform stats
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.3cm}

    \begin{alertblock}{Key Advantage}
        \textbf{Complete labeled dataset} for supervised learning -- rare in prediction markets! \\
        External features (polls, news sentiment) can enhance models.
    \end{alertblock}
\end{frame}
